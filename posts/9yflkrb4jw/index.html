python将PDF每一页导出为600DPI的超清JPG

October 3, 2025 
import fitz  # PyMuPDF库
import os

def pdf_to_high_res_jpg(pdf_path, output_folder, dpi=600):
    """
    将PDF文件的每一页转换为高分辨率JPG图片
    
    参数:
        pdf_path (str): PDF文件路径
        output_folder (str): 输出图片的文件夹
        dpi (int): 图片分辨率，默认为600
    """
    # 创建输出文件夹（如果不存在）
    os.makedirs(output_folder, exist_ok=True)
    
    try:
        # 打开PDF文件
        pdf_document = fitz.open(pdf_path)
        print(f"成功打开PDF文件，共{len(pdf_document)}页")
        
        # 遍历每一页
        for page_number in range(len(pdf_document)):
            # 获取当前页
            page = pdf_document[page_number]
            
            # 设置缩放因子，72DPI是默认值，要得到600DPI需要缩放600/72倍
            zoom = dpi / 72
            mat = fitz.Matrix(zoom, zoom)
            
            # 将页面转换为像素图
            pix = page.get_pixmap(matrix=mat)
            
            # 生成输出文件名（例如：page_001.jpg）
            output_filename = f"page_{page_number + 1:03d}.jpg"
            output_path = os.path.join(output_folder, output_filename)
            
            # 保存为JPG
            pix.save(output_path)
            print(f"已保存: {output_path}")
        
        # 关闭PDF文件
        pdf_document.close()
        print("转换完成！")
        
    except Exception as e:
        print(f"转换过程中出错: {str(e)}")

# 使用示例
if __name__ == "__main__":
    # 输入PDF文件路径
    input_pdf =     pdf_file_path = "First Aid Cases For The USMLE Step 1 (Tao Le, Richard Giovane) (Z-Library).pdf"  # 替换为你的PDF文件路径

    # 输出文件夹路径
    output_dir = "pdf_pages_jpg"
