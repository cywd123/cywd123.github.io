      <h2 class="font-semibold mb-4 text-slate-700"><i class="fa fa-upload text-blue-500 mr-2"></i>上传图片</h2>
      
      <div id="dropArea" class="flex-1 border-2 border-dashed border-slate-300 rounded-lg flex flex-col items-center justify-center p-6 transition-colors cursor-pointer min-h-[200px] relative">
        <input type="file" id="fileInput" accept="image/*" class="hidden">
        
        <div id="uploadPrompt" class="text-center">
          <i class="fa fa-cloud-upload text-4xl text-slate-300 mb-3"></i>
          <p class="text-sm text-slate-500">点击或拖拽图片至此</p>
        </div>

        <img id="previewImg" class="hidden max-h-64 max-w-full object-contain rounded shadow-sm z-10">
        
        <button id="clearBtn" class="hidden absolute top-2 right-2 bg-red-500 text-white w-6 h-6 rounded-full hover:bg-red-600 z-20 items-center justify-center text-xs">
          <i class="fa fa-times"></i>
        </button>

        <div class="flex justify-between text-xs text-slate-500 mb-1">
          <span>正在识别...</span>
          <span id="progressText">0%</span>
        </div>
        <div class="h-2 bg-slate-100 rounded-full overflow-hidden">
          <div id="progressBar" class="h-full bg-blue-500 transition-all duration-300 w-0"></div>

      <button id="runBtn" disabled class="mt-4 w-full py-3 rounded-lg font-medium text-white bg-slate-300 cursor-not-allowed transition-colors flex justify-center items-center">
        开始识别

    <div class="bg-white rounded-xl shadow-md p-6 flex flex-col h-full">
      <div class="flex justify-between items-center mb-4">
        <h2 class="font-semibold text-slate-700"><i class="fa fa-file-text text-emerald-500 mr-2"></i>识别结果</h2>

          placeholder="等待识别..." readonly></textarea>

  <script>
    // 元素引用
    const els = {
      dropArea: document.getElementById('dropArea'),
      fileInput: document.getElementById('fileInput'),
      previewImg: document.getElementById('previewImg'),
      uploadPrompt: document.getElementById('uploadPrompt'),
      clearBtn: document.getElementById('clearBtn'),
      runBtn: document.getElementById('runBtn'),
      progressBox: document.getElementById('progressBox'),
      progressBar: document.getElementById('progressBar'),
      progressText: document.getElementById('progressText'),
      resultBox: document.getElementById('resultBox'),
      copyBtn: document.getElementById('copyBtn'),
      toast: document.getElementById('toast')
    };

    let currentFile = null;

    // --- 工具函数：清理文本空格 ---
    // 1. 去除中文字符之间的空格
    // 2. 将多个空行合并为一个换行
    // 3. 去除每行首尾空格
    function cleanText(text) {
      return text
        .replace(/([^\x00-\xff])\s+([^\x00-\xff])/g, '$1$2') // 去除汉字间空格
        .replace(/[\r\n]+/g, '\n')  // 合并多个换行符
        .split('\n').map(line => line.trim()).filter(line => line.length > 0).join('\n') // 去空行并Trim每一行
        .trim();
    }

    // --- 交互逻辑 ---

    // 拖拽效果
    ['dragenter', 'dragover'].forEach(e => els.dropArea.addEventListener(e, (ev) => {
      ev.preventDefault(); els.dropArea.classList.add('drag-active');
    }));
    ['dragleave', 'drop'].forEach(e => els.dropArea.addEventListener(e, (ev) => {
      ev.preventDefault(); els.dropArea.classList.remove('drag-active');
    }));

    // 文件处理
    const handleFile = (file) => {
      if (!file || !file.type.startsWith('image/')) return alert('请上传图片文件');
      currentFile = file;
      
      const reader = new FileReader();
      reader.onload = (e) => {
        els.previewImg.src = e.target.result;
        els.previewImg.classList.remove('hidden');
        els.clearBtn.classList.remove('hidden');
        els.uploadPrompt.classList.add('hidden');
        
        // 激活按钮
        els.runBtn.disabled = false;
        els.runBtn.className = 'mt-4 w-full py-3 rounded-lg font-medium text-white bg-blue-500 hover:bg-blue-600 transition-colors shadow-md hover:shadow-lg';
        
        // 重置结果区
        els.resultBox.value = '';
        els.copyBtn.disabled = true;
        els.progressBox.classList.add('hidden');
      };
      reader.readAsDataURL(file);
    };

    els.dropArea.addEventListener('drop', (e) => handleFile(e.dataTransfer.files[0]));
    els.dropArea.addEventListener('click', (e) => {
      if(e.target !== els.clearBtn && e.target.parentNode !== els.clearBtn) els.fileInput.click();
    });
    els.fileInput.addEventListener('change', (e) => handleFile(e.target.files[0]));

    // 清除图片
    els.clearBtn.addEventListener('click', (e) => {
      e.stopPropagation();
      currentFile = null;
      els.previewImg.classList.add('hidden');
      els.clearBtn.classList.add('hidden');
      els.uploadPrompt.classList.remove('hidden');
      els.runBtn.disabled = true;
      els.runBtn.className = 'mt-4 w-full py-3 rounded-lg font-medium text-white bg-slate-300 cursor-not-allowed transition-colors';
      els.fileInput.value = '';
    });

    // 执行OCR
    els.runBtn.addEventListener('click', async () => {
      if (!currentFile) return;

      els.progressBox.classList.remove('hidden');
      els.runBtn.disabled = true;
      els.runBtn.classList.add('opacity-75');
      
      try {
        const { data: { text } } = await Tesseract.recognize(currentFile, 'chi_sim', {
          logger: m => {
            if (m.status === 'recognizing text') {
              const pct = Math.floor(m.progress * 100) + '%';
              els.progressBar.style.width = pct;
              els.progressText.innerText = pct;
            }
          }
        });

        // 关键：使用 cleanText 处理结果
        const finalText = cleanText(text);
        
        els.resultBox.value = finalText || '未能识别出文字';
        els.copyBtn.disabled = !finalText;
      } catch (err) {
        els.resultBox.value = '识别出错: ' + err.message;
      } finally {
        els.runBtn.disabled = false;
        els.runBtn.classList.remove('opacity-75');
      }
    });

    // 复制功能
    els.copyBtn.addEventListener('click', () => {
      els.resultBox.select();
      document.execCommand('copy');
      els.toast.classList.remove('opacity-0');
      setTimeout(() => els.toast.classList.add('opacity-0'), 2000);
    });
  </script>
</body>
